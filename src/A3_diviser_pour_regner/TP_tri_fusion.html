class Cell:
    """ Classe cellule pour liste chainée"""
    
    def __init__(self, v, s):
        self.val = v
        self.suiv = s

# TESTS a voir éventuellement sur pythontutor
lst1 = Cell(10, Cell(5, Cell(-4, Cell(0, Cell(1, None)))))
lst2 = Cell(2, lst1)

def taille(l):
    return 1 if l.suiv is None else 1 + taille(l.suiv)

Couper une liste en deux¶

Il s’agit de séparer les éléments d’une liste en deux listes de même taille (à un élément près si la taille de la liste originale est impaire).

Proposition 1¶

Idée:

  • trouver la taille \(n\) de la liste;

  • mettre les \(\left\lfloor \dfrac{n}{2}\right\rfloor\) premiers dans la première liste et les autres dans la seconde liste.

def coupe_v1(l):
    """
    coupe l en deux parties (égales si la longueur de l est paire).
    renvoie deux listes chainées l1 et l2
    """
    # A compléter
    l1, l2 = None, None
    n = taille(l) # il faut la taille de la liste
    cpt = 0
    l_tempo = l
    while l_tempo is not None: # tant qu'on n'est pas en fin de liste
        if cpt < n // 2:
            l1 = Cell(l_tempo.val, l1)
        else:
            l2 = Cell(l_tempo.val, l2)
        l_tempo = l_tempo.suiv
        cpt = cpt + 1
    return l1, l2

# TESTS
l1, l2 = coupe_v1(lst2)
affiche_list(lst2)

Fusionner deux listes¶

On considère deux listes chaînées l1 et l2 que l’on suppose triées par ordre croissant. On demande d’écrire une fonction fusion qui prend en paramètre l1 et l2 et qui renvoie une liste chaînée contenant tous les éléments de l1 et l2 triés par ordre croissant. La récursivité est bienvenue !

def fusion(l1, l2):
    """
    renvoie la fusion ordonée de deux listes triées;
    l1, l2: listes chaînées
    """
    # Les cas de base
    if l1 is None: return l2
    if l2 is None: return l1
    # Cas récursif
    if l1.val < l2.val:
        return Cell(l1.val, fusion(l1.suiv, l2))
    else:
        return Cell(l2.val, fusion(l1, l2.suiv))

# TESTS
x = Cell(3, Cell(8, Cell(13, Cell(17, None))))
y = Cell(-2, Cell(1, Cell(7, Cell(11, None))))
affiche_list(x)
affiche_list(y)
liste_finale = fusion(x, y)
affiche_list(liste_finale)

3  --> 8  --> 13  --> 17  --> None
-2  --> 1  --> 7  --> 11  --> None
-2  --> 1  --> 3  --> 7  --> 8  --> 11  --> 13  --> 17  --> None
